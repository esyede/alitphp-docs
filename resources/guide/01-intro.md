function pageHelper($arg,$type) {
	$app=$this->app;
	if (is_null($arg))
		$app->mset([
			'Html.pageTitle'=>$app->get('PACKAGE').' | Docs',
			'Html.viewFile'=>$type.'-list'
		]);
	else {
		$source=$app->get('ROOT').DS.'markdown'.DS.$type.DS.$arg.'.md';
		if (!file_exists($source))
			$app->redirect('/notfound');
		else $app->mset([
			'Html.pageTitle'=>$app->get('PACKAGE').' | '.ucwords(str_replace('-',' ',$arg)),
			'Html.viewFile'=>'partials/layout',
			'Html.pageContent'=>$this->renderMarkdown($type.'/'.$arg.'.md')
		]);
	}
}
